# Sidebar Component

The Sidebar component provides a comprehensive navigation sidebar with storage meter, user dropdown, and collapsible functionality.

**Location**: `components/sidebar/Sidebar.tsx`

## Basic Usage

```tsx
import Sidebar from '@proton/components/components/sidebar/Sidebar';
import SidebarNav from '@proton/components/components/sidebar/SidebarNav';
import SidebarList from '@proton/components/components/sidebar/SidebarList';
import SidebarListItem from '@proton/components/components/sidebar/SidebarListItem';
import { APPS } from '@proton/shared/lib/constants';

const MySidebar = () => {
  const [expanded, setExpanded] = useState(false);

  return (
    <Sidebar
      app={APPS.PROTONMAIL}
      logo={<Logo />}
      expanded={expanded}
      onToggleExpand={() => setExpanded(!expanded)}
      appsDropdown={<AppsDropdown />}
    >
      <SidebarNav>
        <SidebarList>
          <SidebarListItem>
            <a href="/inbox">Inbox</a>
          </SidebarListItem>
          <SidebarListItem>
            <a href="/sent">Sent</a>
          </SidebarListItem>
        </SidebarList>
      </SidebarNav>
    </Sidebar>
  );
};
```

## Props

Extends `ComponentPropsWithoutRef<'div'>`

<ParamField path="app" type="APP_NAMES" required>
  Application identifier (e.g., APPS.PROTONMAIL)
</ParamField>

<ParamField path="logo" type="ReactNode">
  Logo component to display
</ParamField>

<ParamField path="expanded" type="boolean">
  Whether sidebar is expanded (mobile only). Default: `false`
</ParamField>

<ParamField path="onToggleExpand" type="() => void">
  Callback when sidebar expand/collapse is toggled
</ParamField>

<ParamField path="primary" type="ReactNode">
  Primary action button (e.g., "Compose" button)
</ParamField>

<ParamField path="children" type="ReactNode">
  Sidebar navigation content
</ParamField>

<ParamField path="version" type="ReactNode">
  Version information to display
</ParamField>

<ParamField path="hasAppLinks" type="boolean">
  Whether to show app links. Default: `true`
</ParamField>

<ParamField path="appsDropdown" type="ReactNode | null" required>
  Apps dropdown component
</ParamField>

<ParamField path="preFooter" type="ReactNode">
  Content rendered before storage meter
</ParamField>

<ParamField path="postFooter" type="ReactNode">
  Content rendered after storage meter
</ParamField>

<ParamField path="growContent" type="boolean">
  Whether content grows to fill available space. Default: `true`
</ParamField>

<ParamField path="showStorage" type="boolean">
  Whether to show storage meter. Default: `true`
</ParamField>

<ParamField path="collapsed" type="boolean">
  Whether sidebar is collapsed. Default: `false`
</ParamField>

<ParamField path="navigationRef" type="Ref<HTMLDivElement>">
  Ref for navigation container
</ParamField>

<ParamField path="wavyMeter" type="boolean">
  Use wavy style for storage meter. Default: `false`
</ParamField>

<ParamField path="hasComposerInFocus" type="boolean">
  Used on Mail Desktop App for draggable areas. Default: `false`
</ParamField>

## Storage Meter

The sidebar automatically displays a storage meter showing used/available space. This can be controlled with the `showStorage` prop.

```tsx
<Sidebar
  app={APPS.PROTONMAIL}
  showStorage={true}
  wavyMeter={false}
  // ... other props
>
  {/* Content */}
</Sidebar>
```

## Best Practices

### Mobile Responsiveness

```tsx
const [expanded, setExpanded] = useState(false);

return (
  <>
    {/* Mobile hamburger button */}
    <Button
      className="md:hidden"
      onClick={() => setExpanded(true)}
    >
      <Icon name="hamburger" />
    </Button>

    <Sidebar
      app={APPS.PROTONMAIL}
      expanded={expanded}
      onToggleExpand={() => setExpanded(false)}
      // ... other props
    >
      {/* Navigation */}
    </Sidebar>
  </>
);
```

### Collapsible Sidebar

```tsx
const [collapsed, setCollapsed] = useState(false);

return (
  <div className="flex">
    <Sidebar
      app={APPS.PROTONMAIL}
      collapsed={collapsed}
      logo={collapsed ? <IconLogo /> : <FullLogo />}
      primary={
        collapsed ? (
          <Button icon onClick={handleCompose}>
            <Icon name="pen" />
          </Button>
        ) : (
          <Button fullWidth onClick={handleCompose}>
            Compose
          </Button>
        )
      }
    >
      <SidebarNav>
        <SidebarList>
          <SidebarListItemLink to="/inbox" icon="inbox">
            {!collapsed && 'Inbox'}
          </SidebarListItemLink>
        </SidebarList>
      </SidebarNav>
    </Sidebar>

    <Button
      className="sidebar-toggle"
      onClick={() => setCollapsed(!collapsed)}
    >
      <Icon name={collapsed ? 'chevron-right' : 'chevron-left'} />
    </Button>
  </div>
);
```
